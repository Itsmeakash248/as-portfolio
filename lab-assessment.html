Descriptive Statistics and Histogram

descriptive visualization 
import numpy as np
import matplotlib.pyplot as plt

data = [75, 89, 66, 52, 90, 68, 83, 94, 77, 60]

mean = np.mean(data)
median = np.median(data)
std_dev = np.std(data)

print("Mean:", mean)
print("Median:", median)
print("Standard Deviation:", std_dev)

plt.hist(data, bins=5, color="#0c8a72", edgecolor="white")
plt.title("Histogram of Marks")
plt.xlabel("Marks")
plt.ylabel("Frequency")
plt.show()

PMF and CDF for Defective Chips

distribution 
import math

N = 25   # total chips
K = 5    # defective chips
n = 4    # chips selected
x_values = [0, 1, 2, 3, 4]

pmf = []
for x in x_values:
    p = (math.comb(K, x) * math.comb(N - K, n - x)) / math.comb(N, n)
    pmf.append(p)

cdf = []
running = 0
for p in pmf:
    running += p
    cdf.append(running)

for x, p, c in zip(x_values, pmf, cdf):
    print(x, round(p, 4), round(c, 4))

Mean, Variance, and CDF of a Discrete RV

distribution 
x = [-2, -1, 0, 1, 2, 3]
k = 0.1
p = [0, 0.1, k, 0.2, 2*k, 0.3]

mean = sum(xi * pi for xi, pi in zip(x, p))
ex2 = sum((xi ** 2) * pi for xi, pi in zip(x, p))
variance = ex2 - (mean ** 2)

cdf = []
running = 0
for pi in p:
    running += pi
    cdf.append(running)

print("Mean:", round(mean, 4))
print("Variance:", round(variance, 4))
print("CDF:", [round(val, 4) for val in cdf])

Poisson Event Queries

distribution probability 
import math

lam = 1.0

def poisson(x):
    return (math.exp(-lam) * (lam ** x)) / math.factorial(x)

exactly_three = poisson(3)
at_least_one = 1 - poisson(0)
between_two_and_four = poisson(2) + poisson(3) + poisson(4)

print("P(X=3):", round(exactly_three, 4))
print("P(X>=1):", round(at_least_one, 4))
print("P(2<=X<=4):", round(between_two_and_four, 4))

Frequency Curve with gaussian_kde

visualization descriptive 
import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import gaussian_kde

marks = np.array([75, 89, 66, 52, 90, 68, 83, 94, 77, 60])
density = gaussian_kde(marks)
x = np.linspace(marks.min(), marks.max(), 100)

plt.plot(x, density(x), color="#ef8354")
plt.title("Frequency Curve using gaussian_kde")
plt.xlabel("Marks")
plt.ylabel("Density")
plt.show()
